  • 在 D 盘 JetBrains\plugins 目录下创建一个名为 live_template.groovy 的脚本，代码如下
def version = "1.0.0";
def packageName = _1;
def className = _2;
def methodName = _3;
def jsMethodName = _4;
def methodParameters = _5;
def methodReturnType = _6;
def date = _7;
def time = _8;
def author = "him";

def result = "";
result += "\n";

// 生成方法参数列表
def outputParams = "";
for(param in methodParameters) {
    outputParams += " * @param " + param + "\n";
}

// 生成返回类型
def outputReturnType = "";
if (methodReturnType != "void") {
    outputReturnType = " * @return " + methodReturnType;
}

def outputDescription = "";
outputDescription += " * <p>";
outputDescription += " Description:\n" ;
outputDescription += " * \n";

def outputAuthor = " * @author " + author + "\n";
def outputVersion = " * @version " + version + "\n";
def outputDatetime = " * @since " + date + " " + time + "\n";
def outputPackageName = " * @see " + packageName;

// 作用在方法上
if (methodName) {
	result += outputParams;
	result += outputReturnType;
}

// 作用在 JS 方法上
else if (jsMethodName) {
	
}

// 其它
else {
	result += outputDescription;
	result += outputAuthor;
	result += outputVersion;
	result += outputDatetime;
	result += outputPackageName;
}

return result;

  • Configure -> Plugins -> Marketplace 在输入框中输入插件名 点击 Install 即可安装
