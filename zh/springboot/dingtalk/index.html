# 项目介绍

# 概述

钉钉群机器人推送群消息，封装了下常用的一些场景

# 主要技术

技术 说明
Spring Boot 新一代 JavaEE 开发标准
TkMyBatis 基于 MyBatis 二次开发的轻量级框架，用于简化 MyBatis 操作
MyBatisGenerator Maven 插件，用于 MyBatis 相关代码生成
MybatisCodeHelper Intellij IDEA 插件，用于 MyBatis 相关代码生成
PageHelper MyBatis 分页插件
Swagger API 文档生成工具
HikariCP 数据库连接池

# 通用的返回结果集

package com.swagger.config.common;

import lombok.AllArgsConstructor;
import lombok.Data;
import java.io.Serializable;
import java.util.List;

@Data
public class ResultForm implements Serializable {
    public static final String CODE = "200";
    public static final String ERROR = "error";
    public static final String MESSAGE = "操作成功";

    private String code;
    private Object data;
    private String message;
    private Cursor cursor;
    private List<Error> errors;

    public static ResultForm createSuccessResult() {
        return createResult(CODE, null, MESSAGE, null, null);
    }

    public static ResultForm createSuccessResult(String message) {
        return createResult(CODE, null, message, null, null);
    }

    public static ResultForm createSuccessResult(Object data, String message) {
        return createResult(CODE, data, message, null, null);
    }

    public static ResultForm createSuccessResult(Object data) {
        return createResult(CODE, data, MESSAGE, null, null);
    }

    public static ResultForm createSuccessResult(Object data, Cursor cursor, String message) {
        return createResult(CODE, data, message, cursor, null);
    }

    public static ResultForm createErrorResult(String message) {
        return createResult(ERROR, null, message, null, null);
    }

    public static ResultForm createErrorResult(List<Error> errors) {
        return createResult(ERROR, null, null, null, errors);
    }


    /**
     * @param code
     * @param data
     * @param message
     * @param cursor
     * @param errors
     * @return
     */
    private static ResultForm createResult(String code, Object data, String message, Cursor cursor, List<Error> errors) {
        ResultForm ResultForm = new ResultForm();
        ResultForm.setCode(code);
        ResultForm.setData(data);
        ResultForm.setMessage(message);
        ResultForm.setCursor(cursor);
        ResultForm.setErrors(errors);

        return ResultForm;
    }

    @Data
    public static class Cursor {
        private int total;
        private int offset;
        private int limit;
    }

    @Data
    @AllArgsConstructor
    public static class Error {
        private String field;
        private String message;
    }
}
